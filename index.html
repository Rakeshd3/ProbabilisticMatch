def probabilistic_match(reference_df, unmatched_df):
    indexer = rl.index.SortedNeighbourhood('customer_name', window=5)
    candidate_pairs = indexer.index(reference_df, unmatched_df)

    comparer = rl.Compare()
    comparer.string('customer_name', 'customer_name', method='jaro_winkler')
    comparer.string('city', 'city', method='levenshtein')
    comparer.numeric('zip_code', 'zip_code', method='gauss')

    features = comparer.compute(candidate_pairs, reference_df, unmatched_df)
    return features[features.sum(axis=1) > matching_threshold]
